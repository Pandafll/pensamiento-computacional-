def pares_impares(lista):
    pares=0
    impares=0
    for i in lista:
        if i%2==0:
            pares=pares+1
        else:
            impares=impares+1
    print(f"hay {pares } pares, y {impares} impares")      

def positivo_negativo(lista):
    positivo = 0
    negativo = 0
    ceros = 0
    for i in lista:
        if i > 0:
            positivos = positivos + 1 
        elif i < 0:
            negativos = negativos + 1 
        else:
            ceros = ceros + 1
    print(f"{positivos} positivos, {negativos} negativos {ceros}, ceros") 

def mayor_menor (lista):
    mayor = lista[0]
    menor = lista[0]
    for i in lista:
        if i > mayor:
            mayor = lista[i] 
        elif menor < menor:
            menor = lista[i] 
    print(f"El mayor es {mayor}, y el menor es {menor}")
