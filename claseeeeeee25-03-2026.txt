# Figura 1: Tabla de multiplicación
filas = 4
columnas = 5

for i in range(1, filas + 1):  # Bucle para las filas
    for j in range(1, columnas + 1):  # Bucle para las columnas
        # Multiplica fila * columna. El format {:2d} asegura que cada número ocupe 2 espacios para alinearse.
        print("{:2d}".format(i * j), end="  ")
    print()  # Salto de línea después de cada fila
    
    
    # Figura 2: Triángulo creciente
filas = 3

for i in range(1, filas + 1):
    print("*" * i)
    
    
    
    # Figura 3: Punta de flecha
max_ancho = 4

# Parte creciente (del 1 al 4)
for i in range(1, max_ancho + 1):
    print("*" * i)

# Parte decreciente (del 3 al 1)
# range(inicio, fin, paso). El paso es -1 para ir hacia atrás.
for i in range(max_ancho - 1, 0, -1):
    print("*" * i)
    
    
    
    # Figura 4: Pirámide centrada
filas = 5

for i in range(1, filas + 1):
    # Calcula espacios: más en la primera fila, menos en la última.
    print(" " * (filas - i), end="")
    # Calcula asteriscos: secuencia impar 1, 3, 5, 7, 9.
    print("*" * ( (2 * i) - 1 ) )
    
    
    
    # Figura 5: Pirámide hueca
filas = 5
ancho_base = 9 # (2*5)-1

for i in range(1, filas + 1):
    # Primero los espacios iniciales para centrar
    print(" " * (filas - i), end="")

    # Si es la última fila, imprímela sólida
    if i == filas:
        print("*" * ancho_base)
    # Si no, imprime el borde izquierdo, el hueco, y el borde derecho
    else:
        # Borde izquierdo
        print("*", end="")

        # Hueco central: más ancho según la fila
        # El ancho del hueco es i-1, pero para los asteriscos internos es (2*(i-1))-1
        if i > 1:
            espacios_internos = (2 * (i - 1)) - 1
            print(" " * espacios_internos, end="")
            # Borde derecho
            print("*", end="")

        print() # Salto de línea
    
    
    
    # --- Configuración del Diamante ---
# El 'tamanio' define cuántas filas tendrá la parte superior.
# Para la figura de tu foto (que tiene una fila central con 3 asteriscos), el tamaño es 3.
tamanio = 3 

# Define qué carácter quieres usar. Usaremos el asterisco como en tu foto.
caracter = "*"

# --- PARTE 1: Mitad Superior del Diamante ---
# Este bucle controla las filas desde la 1 hasta el 'tamanio'.
for i in range(1, tamanio + 1):
    
    # 1. IMPRIMIR ESPACIOS (para centrar)
    # Necesitamos imprimir espacios al principio de cada fila.
    # El número de espacios disminuye en cada fila: (tamanio - fila_actual).
    # La operación (" " * numero) multiplica esa cadena de texto.
    print(" " * (tamanio - i), end="")
    
    # 2. IMPRIMIR EL CARÁCTER
    # Ahora imprimimos el asterisco. Entre cada asterisco, añadimos un espacio en blanco
    # para que la figura no se vea "aplastada" verticalmente. Esto crea el efecto de mosaico.
    print((caracter + " ") * i)

# --- PARTE 2: Mitad Inferior del Diamante ---
# Este bucle controla las filas decrecientes, empezando un paso ANTES de la fila más ancha
# y bajando hasta 1. (Inicio, Fin, Paso=-1 para decrecer).
for i in range(tamanio - 1, 0, -1):
    
    # 1. IMPRIMIR ESPACIOS
    # El número de espacios aumenta en cada fila: (tamanio - fila_actual).
    print(" " * (tamanio - i), end="")
    
    # 2. IMPRIMIR EL CARÁCTER
    # Imprimimos el asterisco con un espacio, 'i' veces.
    print((caracter + " ") * i)
